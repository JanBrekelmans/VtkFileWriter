171 pieceNode->InsertEndChild(cellDataElement);

178 void writeCells(tinyxml2::XMLDocument& doc, tinyxml2::XMLElement* cellElement, const std::vector<VtkCell>& cells) {
179 auto connectivity = doc.NewElement("DataArray");
180 cellElement->InsertEndChild(connectivity);
181 connectivity->SetAttribute("type", "Int32");
182 connectivity->SetAttribute("Name", "connectivity");
183 connectivity->SetAttribute("format", "ascii");
184
185 auto offsets = doc.NewElement("DataArray");
186 cellElement->InsertEndChild(offsets);
187 offsets->SetAttribute("type", "Int32");
188 offsets->SetAttribute("Name", "offsets");
189 offsets->SetAttribute("format", "ascii");
190
191 auto types = doc.NewElement("DataArray");
192 cellElement->InsertEndChild(types);
193 types->SetAttribute("type", "Int32");
194 types->SetAttribute("Name", "types");
195 types->SetAttribute("format", "ascii");
196
197 std::string connectivityString, offsetString, typesString;
198 std::size_t cumulativeOffset = 0;
199 for (const VtkCell& cell : cells) {
200 for (const auto& p : cell.points) {
201 connectivityString += std::to_string(p) + ' ';
202 }
203
204 cumulativeOffset += cell.points.size();
205 offsetString += std::to_string(cumulativeOffset) + ' ';
206
207 typesString += std::to_string((int)cell.cellType) + ' ';
208 }
209 connectivityString.pop_back();
210 offsetString.pop_back();
211 typesString.pop_back();
212
213 connectivity->SetText(connectivityString.c_str());
214 offsets->SetText(offsetString.c_str());
215 types->SetText(typesString.c_str());
216 }

218 void writeDataWrappers(tinyxml2::XMLDocument& doc, tinyxml2::XMLElement* xmlElement, const std::vector<DataWrapper*>& dataWrappers) {
219 for (const DataWrapper* const dataWrapper : dataWrappers) {
220 auto dataElement = doc.NewElement("DataArray");

222 dataElement->SetAttribute("Name", dataWrapper->name.c_str());
223 dataElement->SetAttribute("NumberOfTuples", dataWrapper->totalDataPoints);
224 dataElement->SetAttribute("NumberOfComponents", dataWrapper->numberOfComponents);
225 dataElement->SetAttribute("format", "ascii");
226 dataElement->SetText(dataWrapper->contentToString().c_str());
227
228 xmlElement->InsertEndChild(dataElement);
229 }
